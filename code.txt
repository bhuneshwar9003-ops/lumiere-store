        "imports": {

    <script type="module">
        import * as THREE from 'three';

        // === THREE.JS HERO BACKGROUND ===
        const heroCanvas = document.getElementById('hero-canvas');
        const scene = new THREE.Scene();
        const camera = new THREE.PerspectiveCamera(60, window.innerWidth / window.innerHeight, 0.1, 100);
        const renderer = new THREE.WebGLRenderer({ 
            canvas: heroCanvas, 
            antialias: true, 
            alpha: true 
        });

        renderer.setSize(window.innerWidth, window.innerHeight);
        renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));

        const deepSpace = new THREE.Color(0x2D2A26);
        const warmGlow = new THREE.Color(0xF8EDE3);
        const paleHalo = new THREE.Color(0xF8EDE3);

        // Background shader
        const bgGeometry = new THREE.PlaneGeometry(20, 15);
        const bgMaterial = new THREE.ShaderMaterial({
            uniforms: {
                uProgress: { value: 0.0 },
                uTime: { value: 0.0 },
                uColorDark: { value: deepSpace },
                uColorLight: { value: warmGlow }
            },
            vertexShader: `
                varying vec2 vUv;
                void main() {
                    vUv = uv;
                    gl_Position = projectionMatrix * modelViewMatrix * vec4(position, 1.0);
                }
            `,
            fragmentShader: `
                uniform float uProgress;
                uniform float uTime;
                uniform vec3 uColorDark;
                uniform vec3 uColorLight;
                varying vec2 vUv;
                
                void main() {
                    vec2 uv = vUv;
                    
                    float grad = smoothstep(0.0, 1.0, uv.y * 0.6 + 0.2);
                    vec3 base = mix(uColorDark * 0.7, uColorDark, grad);
                    
                    // Warm light from top right
                    float lightX = 1.0 - uv.x;
                    float lightY = uv.y;
                    float lightMask = pow(lightX * lightY, 2.0) * uProgress;
                    
                    // Soft light rays
                    float ray1 = sin(uv.x * 4.0 + uTime * 0.15) * 0.5 + 0.5;
                    float ray2 = sin(uv.y * 3.0 - uTime * 0.1) * 0.5 + 0.5;
                    float rays = ray1 * ray2 * lightMask * 0.15;
                    
                    vec3 light = uColorLight * (lightMask * 0.5 + rays);
                    
                    gl_FragColor = vec4(base + light, 1.0);
                }
            `
        });
        const bgMesh = new THREE.Mesh(bgGeometry, bgMaterial);
        bgMesh.position.z = -5;
        scene.add(bgMesh);

        // Dust particles
        const dustCount = 180;
        const dustPositions = new Float32Array(dustCount * 3);
        const dustSizes = new Float32Array(dustCount);
        const dustPhases = new Float32Array(dustCount);

        for (let i = 0; i < dustCount; i++) {
            dustPositions[i * 3] = (Math.random() - 0.5) * 12;
            dustPositions[i * 3 + 1] = (Math.random() - 0.5) * 8;
            dustPositions[i * 3 + 2] = (Math.random() - 0.3) * 3;
            dustSizes[i] = Math.random() * 4 + 1;
            dustPhases[i] = Math.random() * Math.PI * 2;
        }

        const dustGeometry = new THREE.BufferGeometry();
        dustGeometry.setAttribute('position', new THREE.BufferAttribute(dustPositions, 3));
        dustGeometry.setAttribute('aSize', new THREE.BufferAttribute(dustSizes, 1));
        dustGeometry.setAttribute('aPhase', new THREE.BufferAttribute(dustPhases, 1));

        const dustMaterial = new THREE.ShaderMaterial({
            uniforms: {
                uTime: { value: 0 },
                uProgress: { value: 0 },
                uColor: { value: paleHalo }
            },
            vertexShader: `
                attribute float aSize;
                attribute float aPhase;
                varying float vAlpha;
                uniform float uTime;
                uniform float uProgress;
                
                void main() {
                    float t = uTime * 0.2 + aPhase;
                    vec3 pos = position;
                    pos.x += sin(t) * 0.4;
                    pos.y += cos(t * 0.8) * 0.3;
                    pos.z += sin(t * 0.5) * 0.1;
                    
                    vAlpha = uProgress * 0.7;
                    
                    vec4 mv = modelViewMatrix * vec4(pos, 1.0);
                    gl_PointSize = aSize * uProgress * (200.0 / -mv.z);
                    gl_Position = projectionMatrix * mv;
                }
            `,
            fragmentShader: `
                varying float vAlpha;
                uniform vec3 uColor;
                
                void main() {
                    float d = length(gl_PointCoord - 0.5);
                    if (d > 0.5) discard;
                    float a = smoothstep(0.5, 0.0, d) * vAlpha * 0.5;
                    gl_FragColor = vec4(uColor, a);
                }
            `,
            transparent: true,
            depthWrite: false,
            blending: THREE.AdditiveBlending
        });

        const dust = new THREE.Points(dustGeometry, dustMaterial);
        scene.add(dust);

        // Soft light orbs
        const orbGroup = new THREE.Group();
        const orbCount = 4;

        for (let i = 0; i < orbCount; i++) {
            const orbGeom = new THREE.SphereGeometry(0.15 + Math.random() * 0.1, 16, 16);
            const orbMat = new THREE.MeshBasicMaterial({
                color: paleHalo,
                transparent: true,
                opacity: 0
            });
            const orb = new THREE.Mesh(orbGeom, orbMat);
            orb.position.set(
                2 + Math.random() * 3,
                0.5 + Math.random() * 2,
                -2 - Math.random() * 2
            );
            orb.userData.baseY = orb.position.y;
            orb.userData.speed = 0.3 + Math.random() * 0.3;
            orb.userData.phase = Math.random() * Math.PI * 2;
            orbGroup.add(orb);
        }
        scene.add(orbGroup);

        camera.position.z = 4;

        // Mode tracking
        let scrollProgress = 0;
        let isAway = true;

        const modeLabel = document.getElementById('mode-label');
        const dotAway = document.getElementById('dot-away');
        const dotHome = document.getElementById('dot-home');

        // Animation loop
        let animTime = 0;
        function animate() {
            requestAnimationFrame(animate);
            animTime += 0.016;

            bgMaterial.uniforms.uTime.value = animTime;
            bgMaterial.uniforms.uProgress.value = scrollProgress;

            dustMaterial.uniforms.uTime.value = animTime;
            dustMaterial.uniforms.uProgress.value = scrollProgress;

            // Animate orbs
            orbGroup.children.forEach(orb => {
                orb.position.y = orb.userData.baseY + Math.sin(animTime * orb.userData.speed + orb.userData.phase) * 0.3;
                orb.material.opacity = scrollProgress * 0.15;
            });

            // Subtle camera movement
            camera.position.x = Math.sin(animTime * 0.08) * 0.15;
            camera.position.y = Math.cos(animTime * 0.06) * 0.1;

            renderer.render(scene, camera);
        }
        animate();

        // Scroll handler
        const hero = document.getElementById('hero');
        window.addEventListener('scroll', () => {
            const rect = hero.getBoundingClientRect();
            const h = hero.offsetHeight;
            scrollProgress = Math.max(0, Math.min(1, -rect.top / h + 1));

            const newIsAway = scrollProgress < 0.5;
            if (newIsAway !== isAway) {
                isAway = newIsAway;
                modeLabel.textContent = isAway ? 'Away Mode' : 'Home Mode';
                dotAway.classList.toggle('active', isAway);
                dotHome.classList.toggle('active', !isAway);
            }
        });

        // Resize
        window.addEventListener('resize', () => {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        });

        // === TIME OF DAY DEMO ===
        const timeBtns = document.querySelectorAll('.time-btn');
